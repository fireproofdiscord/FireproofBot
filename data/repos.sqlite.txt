This file contains the table schemas for repos.sqlite

CREATE TABLE "repos" (
	"uuid"	TEXT NOT NULL,
	"slug"	TEXT NOT NULL,
	"guildId"	TEXT NOT NULL UNIQUE,
	PRIMARY KEY("uuid")
)

CREATE TABLE "roles" (
	"uuid"	TEXT NOT NULL,
	"number"	TEXT NOT NULL,
	"roleId"	TEXT NOT NULL,
	"guildId"	TEXT NOT NULL,
	PRIMARY KEY("uuid")
)
